% Problem 1.1
\begin{solution}
  \begin{itemize}
    \item Basis $n = 1$: $\sum_{i=1}^1 i = 1 = \frac{(1 + 1) \cdot 1}{2}$.
    \item Inductive hypothesis: $\sum_{i=1}^n = \frac{n(n+1)}{2}$.
    \item Proof: 
      \begin{align*}
        \sum_{i=1}^{n+1} i &= \sum_{i=1}^n i + (n+1) \\
                           &\eqih \frac{n(n+1)}{2} + (n+1) \\
                           &= \frac{n(n+1)}{2} + \frac{2(n+1)}{2} = \frac{(n+1)(n+2)}{2}
      \end{align*}
  \end{itemize}
\end{solution}
\begin{solution}
  2.3.2 Basis $n = 1$: $1 = t_1 = \sum_{j=1}^1 t_j = \frac{1(1 + 1)(1+2)}{6}$
  Inductive hypothesis: $\sum_{j=1}^n t_j = \frac{n(n+1)(n+2)}{6}$
  Proof:
      \begin{align*}
        \sum_{j=1}^{n+1} t_j &= \sum_{j=1}^n t_j + t_{n+1} \\
            &\eqih \frac{n(n+1)(n+2)}{6} + \frac{(n+1)(n+2)}{2} \\
            &= \frac{(n+1)(n+2)(n+3)}{6}
      \end{align*}

\end{solution}
